//最大公約数を求める
ll gcd(ll a, ll b) {
	if (a % b == 0) {
		return (b);
	}
	else {
		return(gcd(b, a % b));
	}
}
//最小公倍数を求める
ll lcm(ll a, ll b) {
	return a / gcd(a, b) * b;
}
//回文かどうか調べる
bool palindrome(string s) {
	string rs(s.rbegin(), s.rend());
	return s == rs;
}
// スケジューリング問題で使う
using Interval = pair<ll, ll>;
// 区間を終端時刻で大小比較する関数
bool cmp(const Interval &a, const Interval &b) {
	return a.second < b.second;
}
// ABC 097 B
int main() {
	ll X;
	cin >> X;
	ll ans = 1;
	if (X == 1) {
		cout << 1 << endl;
		return 0;
	}
	for (ll i = 2; i*i <= X; i++) {
		ll m = i;
		while (m <= X) {
			ans = max(ans, m);
			m *= i;
		}
	}
	cout << ans << endl;
}
